mod poll_fn;
pub use poll_fn::poll_fn;

cfg_process! {
    mod try_join;
    pub(crate) use try_join::try_join3;
}

    mod block_on;

}

    mod trace;
    pub(crate) use trace::InstrumentedFuture as Future;
}

        pub(crate) use std::future::Future;
    }
